438  ArduJAXRadioGroup(const char* id_base, const char* options[NUM], uint8_t selected_option = 0) : ArduJAXContainer<NUM>(), ArduJAXRadioGroupBase() {
439  for (uint8_t i = 0; i < NUM; ++i) {
440  char* childid = childids[i];
441  strncpy(childid, id_base, ARDUJAX_MAX_ID_LEN-4);
442  itoa(i, &(childid[strlen(childid)]), 10);
443  buttons[i] = ArduJAXCheckButton(childid, options[i], i == selected_option);
444  buttons[i].radiogroup = this;
445  buttonpointers[i] = &buttons[i];
446  }
447  _current_option = selected_option;
448  ArduJAXContainer<NUM>::_children = buttonpointers; // Hm, why do I need to specify ArduJAXContainer<NUM>::, explicitly?
449  _name = id_base;
450  }
452  void selectOptionN(uint8_t num) {
453  for (uint8_t i = 0; i < NUM; ++i) {
454  buttons[i].setChecked(i == num);
455  }
456  _current_option = num; // NOTE: might be outside of range, but that's ok, signifies "none selected"
457  }
459  uint8_t selectedOption() const {
460  return _current_option;
461  }
464  ArduJAXBase* button(uint8_t num) {
465  if (num < NUM) return (&buttons[num]);
466  return 0;
467  }
468 private:
469  ArduJAXCheckButton buttons[NUM];
470  ArduJAXBase* buttonpointers[NUM];
471  char childids[NUM][ARDUJAX_MAX_ID_LEN];
472  int8_t _current_option;
473  void selectOption(ArduJAXCheckButton* which) override {
474  _current_option = -1;
475  for (uint8_t i = 0; i < NUM; ++i) {
476  if (which == buttonpointers[i]) {
477  _current_option = i;
478  } else {
