502 template<size_t NUM> class ArduJAXRadioGroup : public ArduJAXContainer<NUM>, public ArduJAXRadioGroupBase {
503 public:
509  ArduJAXRadioGroup(const char* id_base, const char* options[NUM], uint8_t selected_option = 0) : ArduJAXContainer<NUM>(), ArduJAXRadioGroupBase() {
510  for (uint8_t i = 0; i < NUM; ++i) {
511  char* childid = childids[i];
512  strncpy(childid, id_base, ARDUJAX_MAX_ID_LEN-4);
513  itoa(i, &(childid[strlen(childid)]), 10);
514  buttons[i] = ArduJAXCheckButton(childid, options[i], i == selected_option);
515  buttons[i].radiogroup = this;
516  buttonpointers[i] = &buttons[i];
517  }
518  _current_option = selected_option;
519  ArduJAXContainer<NUM>::_children = buttonpointers; // Hm, why do I need to specify ArduJAXContainer<NUM>::, explicitly?
520  _name = id_base;
521  }
523  void selectOption(uint8_t num) {
524  for (uint8_t i = 0; i < NUM; ++i) {
525  buttons[i].setChecked(i == num);
526  }
527  _current_option = num; // NOTE: might be outside of range, but that's ok, signifies "none selected"
528  }
530  uint8_t selectedOption() const {
531  return _current_option;
532  }
535  ArduJAXBase* button(uint8_t num) {
536  if (num < NUM) return (&buttons[num]);
537  return 0;
538  }
539 private:
540  ArduJAXCheckButton buttons[NUM];
541  ArduJAXBase* buttonpointers[NUM];
542  char childids[NUM][ARDUJAX_MAX_ID_LEN];
543  int8_t _current_option;
544  void selectOption(ArduJAXCheckButton* which) override {
545  _current_option = -1;
546  for (uint8_t i = 0; i < NUM; ++i) {
547  if (which == buttonpointers[i]) {
548  _current_option = i;
549  } else {
550  buttons[i].setChecked(false);
551  }
552  }
