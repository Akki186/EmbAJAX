397  ArduJAXRadioGroup(const char* id_base, const char* options[NUM], uint8_t selected_option = 0) : ArduJAXContainer<NUM>(), ArduJAXRadioGroupBase() {
398  for (uint8_t i = 0; i < NUM; ++i) {
399  char* childid = childids[i];
400  strncpy(childid, id_base, ARDUJAX_MAX_ID_LEN-4);
401  itoa(i, &(childid[strlen(childid)]), 10);
402  buttons[i] = ArduJAXCheckButton(childid, options[i], i == selected_option);
403  buttons[i].radiogroup = this;
404  buttonpointers[i] = &buttons[i];
405  }
406  _current_option = selected_option;
407  ArduJAXContainer<NUM>::_children = buttonpointers; // Hm, why do I need to specify ArduJAXContainer<NUM>::, explicitly?
408  _name = id_base;
409  }
411  void selectOptionN(uint8_t num) {
412  for (uint8_t i = 0; i < NUM; ++i) {
413  buttons[i].setChecked(i == num);
414  }
415  _current_option = num; // NOTE: might be outside of range, but that's ok, signifies "none selected"
416  }
418  uint8_t selectedOption() const {
419  return _current_option;
420  }
423  ArduJAXBase* button(uint8_t num) {
424  if (num < NUM) return (&buttons[num]);
425  return 0;
426  }
427 private:
428  ArduJAXCheckButton buttons[NUM];
429  ArduJAXBase* buttonpointers[NUM];
430  char childids[NUM][ARDUJAX_MAX_ID_LEN];

433  _current_option = -1;
