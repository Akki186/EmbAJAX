471  ArduJAXRadioGroup(const char* id_base, const char* options[NUM], uint8_t selected_option = 0) : ArduJAXContainer<NUM>(), ArduJAXRadioGroupBase() {
472  for (uint8_t i = 0; i < NUM; ++i) {
473  char* childid = childids[i];
474  strncpy(childid, id_base, ARDUJAX_MAX_ID_LEN-4);
475  itoa(i, &(childid[strlen(childid)]), 10);
476  buttons[i] = ArduJAXCheckButton(childid, options[i], i == selected_option);
477  buttons[i].radiogroup = this;
478  buttonpointers[i] = &buttons[i];
479  }
480  _current_option = selected_option;
481  ArduJAXContainer<NUM>::_children = buttonpointers; // Hm, why do I need to specify ArduJAXContainer<NUM>::, explicitly?
482  _name = id_base;
483  }
485  void selectOption(uint8_t num) {
486  for (uint8_t i = 0; i < NUM; ++i) {
487  buttons[i].setChecked(i == num);
488  }
489  _current_option = num; // NOTE: might be outside of range, but that's ok, signifies "none selected"
490  }
492  uint8_t selectedOption() const {
493  return _current_option;
494  }
497  ArduJAXBase* button(uint8_t num) {
498  if (num < NUM) return (&buttons[num]);
499  return 0;
500  }
501 private:
502  ArduJAXCheckButton buttons[NUM];
503  ArduJAXBase* buttonpointers[NUM];
504  char childids[NUM][ARDUJAX_MAX_ID_LEN];
505  int8_t _current_option;
506  void selectOption(ArduJAXCheckButton* which) override {
507  _current_option = -1;
508  for (uint8_t i = 0; i < NUM; ++i) {
509  if (which == buttonpointers[i]) {
510  _current_option = i;
511  } else {
512  buttons[i].setChecked(false);
513  }
514  }
