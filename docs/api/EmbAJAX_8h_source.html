558  EmbAJAXRadioGroup(const char* id_base, const char* options[NUM], uint8_t selected_option = 0) : EmbAJAXContainer<NUM>(), EmbAJAXRadioGroupBase() {

563  buttons[i] = EmbAJAXCheckButton(childid, options[i], i == selected_option);
564  buttons[i].radiogroup = this;
565  buttonpointers[i] = &buttons[i];
566  }
567  _current_option = selected_option;
568  EmbAJAXContainer<NUM>::_children = buttonpointers; // Hm, why do I need to specify EmbAJAXContainer<NUM>::, explicitly?
569  _name = id_base;
570  }
572  void selectOption(uint8_t num) {
573  for (uint8_t i = 0; i < NUM; ++i) {
574  buttons[i].setChecked(i == num);
575  }
576  _current_option = num; // NOTE: might be outside of range, but that's ok, signifies "none selected"
577  }
579  uint8_t selectedOption() const {
580  return _current_option;
581  }

586  return 0;
